function(windeploy target)
  add_custom_command(TARGET ${target}
    POST_BUILD
    COMMAND "${WINDEPLOYQT_EXECUTABLE}" --dir windeploy --compiler-runtime ${CMAKE_CFG_INTDIR}/${target}.exe
	)
endfunction()
